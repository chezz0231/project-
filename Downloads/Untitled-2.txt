class Employee:
    def __init__(self, name, position):
        self.name = name
        self.position = position
    
class Department:
    def __init__(self, name):
        self.name = name
        self.employees = []
    
    def add_employee(self, employee):
        self.employees.append(employee)
    
    def remove_employee(self, employee):
        self.employees.remove(employee)
    
class Project:
    def __init__(self, name, department, employees):
        self.name = name
        self.department = department
        self.employees = employees

employee1 = Employee("John Doe", "Software Engineer")
employee2 = Employee("Jane Smith", "UI/UX Designer")
employee3 = Employee("Mike Johnson", "Project Manager")

department1 = Department("Engineering")
department1.add_employee(employee1)

department2 = Department("Design")
department2.add_employee(employee2)

project1 = Project("Website Redesign", department2, [employee2])
project2 = Project("Mobile App Development", department1, [employee1, employee3])
