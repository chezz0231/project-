from tkinter import *
root = Tk()
root.title("Clicker Game")

clicks = 0

def click_button():
 global clicks
 clicks += 1
 label.config(text="Clicks: " + str(clicks))

label = Label(root, text="Clicks: 0")
label.pack()

button = Button(root, text="Click Me", command=click_button)
button.pack()

root.mainloop()
